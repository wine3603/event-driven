41  void initialise(double width, double height, int Tsize, unsigned int Ssize)
42  {
43  TSleftL.resize(width + 2 * Ssize, height + 2 * Ssize);
44  TSleftH.resize(width + 2 * Ssize, height + 2 * Ssize);
45  TSrightL.resize(width + 2 * Ssize, height + 2 * Ssize);
46  TSrightH.resize(width + 2 * Ssize, height + 2 * Ssize);

48  TSleftL.zero();
49  TSleftH.zero();

51  TSrightH.zero();

57  bool check(int x, int y, int p, int c, int ts)
58  {
59  if(!Ssize) return false;
60 
61  yarp::sig::ImageOf<yarp::sig::PixelInt> *active = 0;
62  if(c == 0) {
63  if(p == 0) {
64  active = &TSleftL;
65  } else if(p == 1) {
66  active = &TSleftH;
67  }
68  } else if(c == 1) {
69  if(p == 0) {
70  active = &TSrightL;
71  } else if(p == 1) {
72  active = &TSrightH;
73  }
74  }
75  if(!active) return false;
76 
77  x += Ssize;
78  y += Ssize;
79 
80 
81  bool add = false;
82  (*active)(x, y) = ts;
83  for(int xi = x - Ssize; xi <= x + Ssize; xi++) {
84  for(int yi = y - Ssize; yi <= y + Ssize; yi++) {
85  int dt = ts - (*active)(xi, yi);
86  if(dt < 0) {
87  dt += vtsHelper::max_stamp;
88  (*active)(xi, yi) -= vtsHelper::max_stamp;
89  }
90  if(dt && dt < Tsize) {
91  add = true;
92  break;
93  }
94  }
95  }
96 
97  return add;
98  }
