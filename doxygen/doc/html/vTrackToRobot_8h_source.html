54  void onRead(vBottle &vBottleIn)
55  {
56 
57  //get the Q
58  vQueue q = vBottleIn.get<GaussianAE>();
59  if(q.empty()) {
60  yWarning() << "q empty in callback function?";
61  return;
62  }
63 
64  //update our current best position of the ball in both cameras
65  bool leftupdated = false, rightupdated = false;
66  vQueue::reverse_iterator qi = q.rbegin();
67  while((!leftupdated || !rightupdated) && qi != q.rend()) {
68  auto v = is_event<GaussianAE>(*qi);
69  if(v->channel == VRIGHT && !rightupdated) {
70  rightupdated = true;
71  rightTarget[0] = v->x;
72  rightTarget[1] = v->y;
73  rightTarget[2] = v->sigx; //radius
74  } else if(v->channel == VLEFT && !leftupdated) {
75  leftupdated = true;
76  leftTarget[0] = v->x;
77  leftTarget[1] = v->y;
78  leftTarget[2] = v->sigx; //radius
79  }
80  qi++;
81  }
82 
83  }
84 
85  void getTargets(yarp::sig::Vector &left, yarp::sig::Vector &right)
86  {
87  left = leftTarget;
88  right = rightTarget;
89  }
90 
91 
92 };
